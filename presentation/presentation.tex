\section{Stage III -- Predictive ML Pipeline}
% ============================================================

% ---- Slide 9: Feature Engineering ----
\begin{frame}{Stage III: Feature Engineering}
\textbf{Custom PySpark Transformers:}
\begin{itemize}
    \item \textbf{SinCosTransformer:} Cyclical encoding of time features
    \begin{itemize}
        \item hour\_of\_day $\rightarrow$ hour\_sin, hour\_cos (period=24)
        \item day\_of\_week $\rightarrow$ day\_sin, day\_cos (period=7)
        \item month $\rightarrow$ month\_sin, month\_cos (period=12)
    \end{itemize}
    \item \textbf{GeoToECEFTransformer:} Geospatial encoding
    \begin{itemize}
        \item lat/lon $\rightarrow$ ECEF x, y, z (WGS84 ellipsoid)
    \end{itemize}
\end{itemize}

\textbf{Pipeline Stages:}
\begin{enumerate}
    \item SinCos $+$ ECEF transformers
    \item StringIndexer + OneHotEncoder (4 categoricals)
    \item VectorAssembler $\rightarrow$ VarianceThresholdSelector
    \item Train/Test split: 80/20 (801k / 200k records)
\end{enumerate}
\end{frame}

\section{Risk Assessment}
% ============================================================

% ---- Slide 13: Risk Assessment ----
\newcommand{\riskframe}{Risk Assessment \& Model Limitations}

\begin{frame}{\riskframe}
\textbf{Potential Risks:}
\begin{enumerate}
    \item \textbf{Temporal drift:} Crime patterns evolve; model trained on 2001--2026 may not generalize to future years
    \item \textbf{Geographic bias:} Some districts have very few records -- predictions may be unreliable
    \item \textbf{Class imbalance:} Model may over-predict ``no arrest'' (74.9\% majority class)
\end{enumerate}

\vspace{0.3cm}
\textbf{Mitigation Strategies:}
\begin{itemize}
    \item Periodic retraining with new data
    \item District-specific threshold tuning
    \item AUC-PR optimization (already addresses class imbalance)
    \item Monitor prediction distribution for drift detection
\end{itemize}
\end{frame}

\section{Challenges Faced}
% ============================================================

% ---- Slide 14: Challenges ----
\begin{frame}{Challenges Faced}
\begin{itemize}
    \item \textbf{Cluster resource limits:} YARN queue contention with other teams
    \item \textbf{Date format conversion:} Sqoop stores timestamps as BIGINT epoch -- required \texttt{from\_unixtime()} conversion
    \item \textbf{Hive strict mode:} Dynamic partitioning required \texttt{nonstrict} mode configuration
\end{itemize}
\end{frame}
